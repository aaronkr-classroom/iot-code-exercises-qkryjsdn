class SayDays:
    
    def __init__(self, year, month, day):
        self.year = year
        self.month = month
        self.day = day

    def is_leap(self):
        if self.year % 400 == 0:
            return True
        elif self.year % 100 == 0:
            return False
        elif self.year % 4 == 0:
            return True
        else:
            return False

    def days(self):
        if self.is_leap():
            month_days = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        else:
            month_days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]

        total = 0
        i = 0
        while i < self.month - 1:
            total = total + month_days[i]
            i = i + 1

        total = total + self.day
        return total

    def days_left(self):
        if self.is_leap():
            return 366 - self.days()
        else:
            return 365 - self.days()

    def week(self):
        y = self.year
        m = self.month
        d = self.day

        if m == 1:
            m = 13
            y = y - 1
        elif m == 2:
            m = 14
            y = y - 1

        k = y % 100
        j = y // 100

        h = (d + ((13 * (m + 1)) // 5) + k + (k // 4) + (j // 4) + (5 * j)) % 7
        return h
    
    def week_name(self):
        names = ["토요일", "일요일", "월요일", "화요일", "수요일", "목요일", "금요일"]
        return names[self.week()]

while True:
    s = input("날짜 입력 (예: 2026 4 8, 종료:a): ")

    if s == "a":
        break

    data = s.split()

    if len(data) != 3:
        print("입력 형식이 잘못되었습니다.")
        print()
        continue

    year = int(data[0])
    month = int(data[1])
    day = int(data[2])

    sd = SayDays(year, month, day)

    print("당해년도 1월 1일부터 몇째 날:", sd.days())
    print("당해년도 12월 31일까지 남은 일수:", sd.days_left())
    print("요일 숫자:", sd.week())
    print("요일 이름:", sd.week_name())
    print()
